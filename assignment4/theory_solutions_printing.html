class PacRectangle:
    def __init__(self, delta=0.05, dist="uniform"):
        """
        Initialize parameters for the pac rectangle study 
        
        :param delta: the parameter for the confidence 
        :param dist: string indicating what distribution to sample data from 
        """
        self.delta = delta
        self.dist = dist 
        
    def genConcept(self):
        """
        generate a random concept rectangle 
        """
        xa, xb = tuple(np.sort(np.random.uniform(0,100,size=2)))
        ya, yb = tuple(np.sort(np.random.uniform(0,100,size=2)))
        self.c = (xa, xb, ya, yb)
        
    def unifData(self, m):
        """
        sample data from uniform distribution 
        
        :param m: number of examples to sample
        """
        self.x_train = np.random.uniform(0,100,size=(m,2))
        self.y_train = np.array([1 if (self.c[0] <= x[0]) and (x[0] <= self.c[1]) and (self.c[2] <= x[1]) and (x[1] <= self.c[3]) else 0 for x in self.x_train], dtype=int)
        self.x_gen = np.random.uniform(0,100,size=(1000,2))
        self.y_gen = np.array([1 if (self.c[0] <= x[0]) and (x[0] <= self.c[1]) and (self.c[2] <= x[1]) and (x[1] <= self.c[3]) else 0 for x in self.x_gen], dtype=int)
        
    def normData(self, m):
        """
        sample data from normal distribution 
        
        :param m: number of examples to sample
        """
        self.x_train = np.random.normal(50,25,size=(m,2))
        self.y_train = np.array([1 if (self.c[0] <= x[0]) and (x[0] <= self.c[1]) and (self.c[2] <= x[1]) and (x[1] <= self.c[3]) else 0 for x in self.x_train], dtype=int)
        self.x_gen = np.random.normal(50,25,size=(1000,2))
        self.y_gen = np.array([1 if (self.c[0] <= x[0]) and (x[0] <= self.c[1]) and (self.c[2] <= x[1]) and (x[1] <= self.c[3]) else 0 for x in self.x_gen], dtype=int)
        
    def getHypothesis(self):
        """
        learn consistent hypothesis rectangle from training data 
        """
        if np.sum(self.y_train==1) > 0:
            xa, xb = np.min(self.x_train[self.y_train==1,0]),np.max(self.x_train[self.y_train==1,0])
            ya, yb = np.min(self.x_train[self.y_train==1,1]),np.max(self.x_train[self.y_train==1,1])
            self.h = (xa, xb, ya, yb)
        else:
            self.h = (-1,-1,-1,-1)
            
    def getLabels(self, rule, X):
        """
        label data using concept rectangle 
        
        :param rule: the concept rectangle 
        :param X: data to label 
        """
        return np.array([1 if (rule[0] <= x[0]) and (x[0] <= rule[1]) and 
                         (rule[2] <= x[1]) and (x[1] <= rule[3]) else 0 for x in X], dtype=int)

    def generalizationSim(self, m):
        """
        simulation to estimate generalization error over many training sets and
        many concept rectangles 
        
        :param m: number of training examples 
        """
        gen_errors = [] 
        num_runs = 100
        for run in range(num_runs):
            # get true concept C 
            self.genConcept() 
            # get training data 
            if self.dist == "uniform": 
                self.unifData(m)
            if self.dist == "normal":
                self.normData(m)
            # get consistent learner
            self.getHypothesis()
            # generalization error 
            gen_err = np.sum(self.getLabels(self.c, self.x_gen) != self.getLabels(self.h, self.x_gen))/len(self.x_gen)
            gen_errors.append(gen_err)
        gen_errors = np.sort(gen_errors)[::-1]
        return gen_errors[int(self.delta*num_runs)-1]
    
    def generalizationStudy(self, ms):
        """
        run simulation for many training set sizes and produce m vs gen_error plot 
        for empirical simulation as well as theoretical bound 
        
        :param ms: an array of training set sizes to test 
        """
        gen_errors = [] 
        for m in ms:
            gen_errors.append(self.generalizationSim(m))
        fig, ax = plt.subplots(nrows=1, ncols=1, figsize=(10,6))
        ax.loglog(ms, gen_errors, marker="o", color="steelblue", label="empirical bounds")
        ax.loglog(ms, [(4./m)*np.log(4/self.delta) for m in ms], marker="o", color="#a76c6e", label="theoretical bounds")
        ax.grid(alpha=0.25, which="both")
        ax.set_xlabel("training set size", fontsize=16)
        ax.set_ylabel("generalization error", fontsize=16)
        ax.set_title(self.dist + " data", fontsize=20)
        ax.legend(loc="upper right")
